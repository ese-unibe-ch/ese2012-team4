    def self.send_mail_to(to,contents)
    require 'rubygems'
    require 'tlsmail'
    require 'time'
    from = 'tradingsystem.mail@gmail.com'
    pw = 'trade1234'

    content = "From: \#{from}\nTo: \#{to}\nsubject: Your Item has been sold!\nDate: \#{Time.now.rfc2822}\n\n    \#{contents}\n\nRegards,\nThe Trading System\n"

    Net::SMTP.enable_tls(OpenSSL::SSL::VERIFY_NONE)
    Net::SMTP.start('smtp.gmail.com', 587, 'gmail.com', from, pw, :login) do |smtp|
      smtp.send_message(content, from, to)
    end
  end
