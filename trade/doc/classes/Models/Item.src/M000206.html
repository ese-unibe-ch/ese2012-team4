    def self.created( name, price, owner, quantity, description = "", image = "")
      item = self.new
      item.price = price
      item.active = false

      item.id = @@count + 1
      item.name = name
      item.owner = owner
      item.quantity = quantity
      item.description = description
      item.image = image
      item.wishlist_users = Array.new
      item.timestamp = Time.now.to_i
      item.head_comments = []
      item.auction = false
      item.permanent = false
      item.currency = "credits"
      item.category = Category.get_supercategory
      item
    end
