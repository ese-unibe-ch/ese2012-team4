    def get_offers(viewer = nil, options ={})
      ret_array = Array.new
      for offer in Offer.get_all(viewer, options)
        if offer.category.name == self.name
          ret_array.push(offer)
        end
      end
      for cat in @subcategories
        ret_array.concat(cat.get_offers)
      end
      ret_array
    end
