  def test_get_all
    user1 = User.created("user1", "password1", "user1@mail.com")
    user1.save

    user2 = User.created("user2", "password2", "user2@mail.com")
    user2.save

    assert(User.get_all(user1).include?(user2))
    assert(!User.get_all(user1).include?(user1))
  end
